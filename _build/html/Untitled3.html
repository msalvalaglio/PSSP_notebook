import pygame
import sys
from IPython import display
import numpy as np

# Initialize Pygame
pygame.init()

# Set up the display
WIDTH, HEIGHT = 640, 480
screen = pygame.Surface((WIDTH, HEIGHT))

# Colors
WHITE = (255, 255, 255)
BLUE = (0, 0, 255)
YELLOW = (255, 255, 0)
RED = (255, 0, 0)

# Player properties
player_width = 40
player_height = 60
player_x = 50
player_y = HEIGHT - player_height
player_speed = 5
jump_speed = -15
gravity = 0.8

# Platform properties
platforms = [
    pygame.Rect(100, 400, 200, 20),
    pygame.Rect(400, 300, 200, 20),
    pygame.Rect(200, 200, 200, 20),
]

# Coin properties
coins = [
    pygame.Rect(150, 370, 20, 20),
    pygame.Rect(450, 270, 20, 20),
    pygame.Rect(250, 170, 20, 20),
]

player_velocity_y = 0
score = 0

clock = pygame.time.Clock()

def update_game():
    global player_x, player_y, player_velocity_y, score

    keys = pygame.key.get_pressed()
    if keys[pygame.K_LEFT]:
        player_x -= player_speed
    if keys[pygame.K_RIGHT]:
        player_x += player_speed

    player_x = max(0, min(player_x, WIDTH - player_width))

    player_velocity_y += gravity
    player_y += player_velocity_y

    player_rect = pygame.Rect(player_x, player_y, player_width, player_height)

    for platform in platforms:
        if player_rect.colliderect(platform) and player_velocity_y > 0:
            player_y = platform.top - player_height
            player_velocity_y = 0

    if player_y >= HEIGHT - player_height:
        player_y = HEIGHT - player_height
        player_velocity_y = 0

    if keys[pygame.K_SPACE] and player_velocity_y == 0:
        player_velocity_y = jump_speed

    for coin in coins[:]:
        if player_rect.colliderect(coin):
            coins.remove(coin)
            score += 1

def draw_game():
    screen.fill(WHITE)
    pygame.draw.rect(screen, RED, (player_x, player_y, player_width, player_height))
    
    for platform in platforms:
        pygame.draw.rect(screen, BLUE, platform)
    
    for coin in coins:
        pygame.draw.rect(screen, YELLOW, coin)
    
    font = pygame.font.Font(None, 36)
    score_text = font.render(f"Score: {score}", True, (0, 0, 0))
    screen.blit(score_text, (10, 10))

def game_loop():
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            pygame.quit()
            sys.exit()

    update_game()
    draw_game()
    
    # Convert the game screen to a displayable format
    frame = pygame.surfarray.array3d(screen)
    return np.ascontiguousarray(frame.swapaxes(0, 1))

def run_game(num_frames=1000):
    for _ in range(num_frames):
        frame = game_loop()
        display.clear_output(wait=True)
        display.display(display.Image(data=frame, format='png'))
        clock.tick(30)

# Run the game
run_game()
